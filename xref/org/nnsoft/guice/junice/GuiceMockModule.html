31  import com.google.common.collect.HashMultimap;
32  import com.google.common.collect.Multimap;
33  import com.google.inject.AbstractModule;
34  import com.google.inject.TypeLiteral;
35  import com.google.inject.name.Names;

37  /**
38   *  <p>This class creates the binding for all mock objects found.</p>
39   *
40   *  <p>Method {@link GuiceMockModule#configure()} creates a binding for each 
41   *  {@link Mock} annotation found. The binding will be created <b>if and only if</b> there is no
42   *  types conflict between {@link Mock} caught.<p>
43   *
44   *  <p><b>A type conflict</b> is detected if two or more field are annotated with the same {@link Mock}
45   *  and no different {@link Mock#annotatedWith} parameter are specified, or two o more equals
46   *  {@link Mock#annotatedWith} parameter are specified for the same type field.</p>
47   *
48   *  <p>If a conflict is detected the binding will not create for the conflicted type,
49   *  moreover the field will be injected into the test class.</p>
50   *
51   * @author Marco Speranza
52   * @version $Id: GuiceMockModule.java 000 2009-12-01 00:00:00Z marco.speranza79 $
53   */
54  public class GuiceMockModule extends AbstractModule {
55  
56      final static private Log logger = LogFactory.getLog(GuiceMockModule.class);
57  
58      final Map<Field, Object> mockedFields;
59  
60      /**
61       * Costructor.
62       * 
63       * @param mockedFields the map of mock fileds.
64       */
65       
66      public GuiceMockModule(final Map<Field, Object> mockedFields) {
67          this.mockedFields = mockedFields;
68      }
69  
70      @SuppressWarnings("unchecked")
71      @Override
72      protected void configure() {
73          final Multimap<Type, Field> fieldsByType = HashMultimap.create();
74  
75          for (final Entry<Field, Object> entry : this.mockedFields.entrySet()) {
76              fieldsByType.put(entry.getKey().getGenericType(), entry.getKey());
77          }
78  
79          for (final Type type : fieldsByType.keySet()) {
80              final Collection<Field> fields = fieldsByType.get(type);
81  
82              boolean isTypeConflicts = false;
83              if (fields.size() != 1) {
84                  isTypeConflicts = checkTypeConflict(fields);
85              }
86  
87              if (!isTypeConflicts) {
88                  for (final Field field : fields) {
89                      final TypeLiteral literal = TypeLiteral.get(type);
90                      final Mock annoBy = field.getAnnotation(Mock.class);
91                      final Object mock = this.mockedFields.get(field);
92                      if (annoBy.annotatedWith() != Mock.NoAnnotation.class) {
93                          bind(literal).annotatedWith(annoBy.annotatedWith()).toInstance(mock);
94                      } else if (!"".equals(annoBy.namedWith())) {
95                          bind(literal).annotatedWith(Names.named(annoBy.namedWith())).toInstance(mock);
96                      } else {
97                          bind(literal).toInstance(mock);
98                      }
99                      if (logger.isDebugEnabled()) {
100                         logger.debug("    Created binding for: " + type + " " + annoBy);
101                     }
102                 }
103             } else {
104                 
105                 final String msg = "   Found multiple annotation @" 
106                     + Mock.class.getSimpleName() 
107                     + " for type: "
108                     + type 
109                     + " Binding skipped!.";
110                 logger.debug(msg);
111                 throw new IllegalArgumentException(msg);
112             }
113         }
114     }
115 
116     /**
117      * 
118      *
119      * @param fields
120      * @return
121      */
122     private boolean checkTypeConflict(Collection<Field> fields) {
123         final List<Class<?>> listAnnotatedType = new ArrayList<Class<?>>();
124         final List<String> listNamedType = new ArrayList<String>();
125         int numOfSimpleType = 0;
126         
127         for (Field field : fields) {
128             final Mock annoBy = field.getAnnotation(Mock.class);
129             
130             if ( annoBy.annotatedWith() == Mock.NoAnnotation.class && "".equals(annoBy.namedWith()) ) {
131                 numOfSimpleType++;       
132             } 
133             if (numOfSimpleType > 1) {
134                 logger.debug("Found multiple simple type");
135                 return true;
136             }
137             
138             if (annoBy.annotatedWith() != Mock.NoAnnotation.class) {
139                 if ( ! listAnnotatedType.contains(annoBy.annotatedWith()) ) {
140                     listAnnotatedType.add(annoBy.annotatedWith());
141                 } else {
142                     //found two fields with same annotation
143                     logger.debug("Found multiple annotatedBy type");
144                     return true;
145                 }
146             }
147             
148             if (! "".equals(annoBy.namedWith())) {
149                 if ( ! listNamedType.contains(annoBy.namedWith()) ) {
150                     listNamedType.add(annoBy.namedWith());
151                 } else {
152                     //found two fields with same named annotation
153                     logger.debug("Found multiple namedWith type");
154                     return true;
155                 }
156             }
157         }
158         return false;
159     }
