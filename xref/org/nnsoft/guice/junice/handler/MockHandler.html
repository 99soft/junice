33  /**
34   * Handler class to handle all {@link Mock} annotations.
35   *
36   * @see ClassVisitor
37   * @see Mock
38   * @author Marco Speranza
39   * @version $Id: MockHandler.java 264 2010-10-07 20:58:47Z marco.speranza79 $
40   */
41  public final class MockHandler implements FieldHandler<Mock> {
42  
43      private static final Log logger = LogFactory.getLog(MockHandler.class);
44  
45      final private HashMap<Field, Object> mockedObjects = new HashMap<Field, Object>(1);
46  
47      /**
48       * 
49       *
50       * @param engine
51       * @return
52       */
53      public HashMap<Field, Object> getMockedObject(MockEngine engine) {
54          createMockedObjectBymockFramekork(engine);
55          return this.mockedObjects;
56      }
57  
58      private void createMockedObjectBymockFramekork(MockEngine engine) {
59          for (Entry<Field, Object> entry : mockedObjects.entrySet()) {
60              if ( entry.getValue() instanceof Class<?>){
61                  Field field = entry.getKey();
62                  Mock mock = field.getAnnotation(Mock.class);
63                  this.mockedObjects.put(entry.getKey(), engine.createMock( (Class<?>) entry.getValue(), mock.type()));
64              }
65          }
66      }
67  
68      @SuppressWarnings("unchecked")
69      public void handle(final Mock annotation, final Field element) throws HandleException {
70          final Class<? super Object> type = (Class<? super Object>) element.getDeclaringClass();
71  
72          if (logger.isDebugEnabled()) {
73              logger.debug("      Found annotated field: " + element);
74          }
75          if (annotation.providedBy().length() > 0 ) {
76                  Class<?> providedClass = type;
77                  if (annotation.providerClass() != Object.class ) {
78                      providedClass = annotation.providerClass();
79                  }
80              try {
81                  Method method = providedClass.getMethod(annotation.providedBy());
82  
83                  if (!element.getType().isAssignableFrom(method.getReturnType())) {
84                      throw new HandleException("Impossible to mock "
85                              + element.getDeclaringClass().getName()
86                              + " due to compatibility type, method provider "
87                              + providedClass.getName() + "#"
88                              + annotation.providedBy() + " returns "
89                              + method.getReturnType().getName());
90                  }
91                  try {
92                      Object mocked = getMockProviderForType(element.getType(),
93                              method, type);
94                      this.mockedObjects.put(element, mocked);
95                  } catch (Throwable t) {
96                      throw new HandleException("Impossible to mock "
97                              + element.getDeclaringClass().getName()
98                              + ", method provider "
99                              + providedClass.getName() + "#"
100                             + annotation.providedBy() + " raised an error", t);
101                 }
102             } catch (SecurityException e) {
103                 throw new HandleException("Impossible to mock "
104                         + element.getDeclaringClass().getName()
105                         + ", impossible to access to method provider "
106                         + providedClass.getName() + "#"
107                         + annotation.providedBy(), e);
108             } catch (NoSuchMethodException e) {
109                 throw new HandleException("Impossible to mock "
110                         + element.getDeclaringClass().getName()
111                         + ", the method provider "
112                         + providedClass.getName() + "#"
113                         + annotation.providedBy() + " doesn't exist.");
114             }
115         } else {
116             this.mockedObjects.put(element, element.getType());
117         }
118     }
119 
120     @SuppressWarnings("unchecked")
121     private <T> T getMockProviderForType(T t, Method method, Class<?> cls)
122             throws HandleException {
123         if (method.getReturnType() == t) {
124             try {
125                 if (logger.isDebugEnabled()) {
126                     logger.debug("        ...invoke Provider method for Mock: "
127                             + method.getName());
128                 }
129                 if (!Modifier.isPublic(method.getModifiers())
130                         || !Modifier.isStatic(method.getModifiers())) {
131                     throw new HandleException("Impossible to invoke method "
132                             + cls 
133                             + "#" + method.getName()
134                             + ". The method shuld be 'static public "
135                             + method.getReturnType().getName()
136                             + " " + method.getName() + "()'");
137                 }
138 
139                 return (T) method.invoke(cls);
140             } catch (Exception e) {
141                 throw new RuntimeException(e);
142             }
143         }
144         throw new HandleException("The method: " + method
145                 + " shuld be return a type " + t);
146     }
